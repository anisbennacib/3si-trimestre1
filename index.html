Fonction suppr_espaces(ch : chaîne) : chaîne
Début

    Tant que Pos("  ", ch) ≠ -1 Faire
        ch ← Effacer(ch, Pos("  ", ch), Pos("  ", ch) + 1)
    Fin Tant que
    { Supprimer espace de début si présent }

        ch ← Effacer(ch, 0, 1)
    FinSi

    FinSi
    Retourner ch
Fin
🐍 Corrigé — Python 
def suppr_espaces(ch):
    # Supprimer les espaces doubles consécutifs
    while ch.find("  ") != -1:
        p = ch.find("  ")
        ch = ch[:p] + ch[p+1:]
    # Supprimer espace de début si présent
    if len(ch) > 0 and ch[0] == " ":
        ch = ch[1:len(ch)]
    # Supprimer espace de fin si présent
    if len(ch) > 0 and ch[len(ch)-1] == " ":
        ch = ch[:len(ch)-1]
    return ch

# 1ère méthode : tri à bulles classique
def Tri_Bulles(T, n):
    permut = True
    while permut:
        permut = False
        for i in range(n-1):
            if T[i] > T[i+1]:
                aux = T[i]
                T[i] = T[i+1]
                T[i+1] = aux
                permut = True

# 2ème méthode : tri à bulles optimisé (n = n-1 à chaque passe)
def Tri_Bulles_Opt(T, n):
    permut = True
    while permut and n > 1:
        permut = False
        for i in range(n-1):
            if T[i] > T[i+1]:
                aux = T[i]
                T[i] = T[i+1]
                T[i+1] = aux
                permut = True
        n = n - 1
